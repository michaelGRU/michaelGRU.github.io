#----------------------------------Intro------------------------------------------------------#

#The program can be used to study various properties of github social networks, such as 
#programmer's collaboration habits, how the collaborated projects vary by programming languages
#and the key programmers who contributed to a repository.

#Unlike traditional social media platforms like Twitter or Facebook, Github is a 
#social networking site that targets software developers. Github offers its users 
#the opportunity to collaborate and contribute to repositories.

#---------------------------------Interpretation----------------------------------------------#

#I included two example networks at the end of the script. The first network shows a weak 
#tie between collaborators. The two programmers don't have a lot in common, and they primarily 
#work on different projects in different languages. The second network is a popular repository. 
#You can still observe cliques and clustering. It is interesting to note that JavaScript is the 
#language that most collaborators use in this repository instead of Python, 
# even though the repository is designed for Python learning. 

#---------------------------------Social Data-------------------------------------------------#

#What social data does the script track? 
#
#Repositories(repo): A repo contains a user's project's files 
#
#Contributors: A person can be granted as a github "contributor" once a pull request
#is merged
#
#Starred repositories: Github users can star repo and topics to keep track of projets
#they find interesting 
#
#programming languages: one repo can be associated with more than one programming language

#----------------------------------Color Code---------------------------------------------#

#NODES COLOR CODES
# black/grey: programming languages
# blue: project contributor 
# purple: other starred repo by the contributors
# red: the repo contributors work on 

# EDGES 
# Contributors (blue nodes) "contribute" to the the shared repo (red node)
# Contributors (blue nodes) "star" the starred repos (purple nodes)
# Repos (red or purple nodes) are "written in" programming languages (black nodes)

#-----------------------------------------------------------------------------------------#

#Other entities that are tracked but not explicitly included in the script: 
#
#github followers
#github outside collobrators 
#github stargazers 

#-----------------------------------------------------------------------------------------#

#create pandas dataframe based on the graph 
def graph_to_pd(G):
    return (pd.DataFrame(dict(
            degree = dict(G.degree),
            deg_centrality = nx.degree_centrality(G),
            closeness= nx.closeness_centrality(G),
            betweenness = nx.betweenness_centrality(G),
            katz = nx.katz_centrality_numpy(G))
        ).round(3)
        .sort_values("degree", ascending=False)
        .reset_index(level=0)
        .rename(columns={'index': 'nodes'})
        .assign(type=lambda x: x.nodes.str.extract(r'\((.*?)\)'))
        .fillna('language')
        
    )

def client_access(user: str):
    """github client access
    Args:
        user (str): username of the target object
    Returns:
        user_access (str): access to the user information
    """

        access = Github(GITHUB_TOKEN) 
        user_access = access.get_user(user)
    except AssertionError:
        print("please enter a valid username")
    return user_access

def draw_network(G):
    #set node color and size
    color_map, size = [], []
    #format labels 
    labels = {}
    #add node color 
    for node in G.nodes(data=True):
        if node[1]['type'] == 'repo':
            color_map.append('red')
            labels[node[0]] = node[0]
        elif node[1]['type'] == 'language':
            color_map.append('black')
            labels[node[0]] = node[0]
        elif node[1]['type'] == 'contributor':
            color_map.append('blue')
            labels[node[0]] = node[0]
        else:
            color_map.append('purple')

    #graph layout
    #pos = nx.spring_layout(G, k=5/math.sqrt(G.order()))
    pos = nx.kamada_kawai_layout(G)
    
    nx.draw_networkx_nodes(G, pos, alpha=0.4, 
                            node_color=color_map, 
                            node_size=[degree * 100 for degree in dict(G.degree).values()])
    nx.draw_networkx_edges(G, pos)
    nx.draw_networkx_labels(G, pos, labels)
    plt.tight_layout()
    plt.gca().margins(0.01, 0.01)
    plt.show()

contributors = [contributor for contributor in repo.get_contributors()]
#store a list of lanauges used to write the repo
languages = [language for language in repo.get_languages()]

#Add the first repo to the graph
G.add_node(repo.name + '(repo)', type='repo', lang=repo.language)

#Add the language associated with the initial collaborated repo
for language in languages: 
    G.add_node(language, type='language')
    G.add_edge(language, repo.name + '(repo)')

#Add contributors 
for contributor in contributors: 
    G.add_node(contributor.login + '(user)', type='contributor')
    G.add_edge(contributor.login + '(user)', repo.name + '(repo)')
    for star_repo in contributor.get_starred():
        if star_repo.name != repo.name: 
            G.add_node(star_repo.name + '(repo)', type='star_repo', lang=star_repo.language)
            G.add_edge(contributor.login + '(user)', star_repo.name + '(repo)')
        
#add languages associated with other starred repos 
for node in list(G.nodes(data=True)): 
    for key, value in node[1].items():
        if node[1][key] is None:
            node[1][key] = 'DNA'
    if node[1]['type'] == 'star_repo':
        G.add_node(node[1]['lang'], type="language")
        G.add_edge(node[0], node[1]['lang'])
